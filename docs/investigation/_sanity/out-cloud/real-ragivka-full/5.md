```go
package ragivka

import (
	"context"
	"database/sql"
	"encoding/json"
	"io"
	"time"

	"github.com/jackc/pgx/v5"
	"github.com/jackc/pgx/v5/pgxpool"
)

type TenantID string
type UserID string
type SessionID string
type MessageID string
type DocumentID string
type ChunkID string
type ArtifactID string
type JobID string
type IdempotencyKey string
type PromptName string
type PromptVersion string
type ToolName string
type APIKey string

type OrchestrationTier int

const (
	L0Deterministic OrchestrationTier = iota
	L1ToolAssistant
	L2WorkflowPipeline
	L3MultiAgentGraph
)

type FSMState string

const (
	StateActive          FSMState = "active"
	StateWaitingForHuman FSMState = "waiting_for_human"
	StateCompleted       FSMState = "completed"
	StateExpired         FSMState = "expired"
)

type ToolPermission string

const (
	PermissionRead  ToolPermission = "read"
	PermissionDraft ToolPermission = "draft"
	PermissionWrite ToolPermission = "write"
)

type LLMProvider string

const (
	ProviderOpenAI    LLMProvider = "openai"
	ProviderAnthropic LLMProvider = "anthropic"
	ProviderOpenRouter LLMProvider = "openrouter"
	ProviderGemini     LLMProvider = "gemini"
	ProviderOllama     LLMProvider = "ollama"
)

type IngestionStatus string

const (
	StatusPending  IngestionStatus = "pending"
	StatusIndexed  IngestionStatus = "indexed"
	StatusStale    IngestionStatus = "stale"
	StatusFailed   IngestionStatus = "failed"
)

type Tenant struct {
	ID        TenantID
	Name      string
	CreatedAt time.Time
	Config    json.RawMessage
}

type User struct {
	ID        UserID
	TenantID  TenantID
	Channel   string
	ChannelID string
	Metadata  json.RawMessage
	CreatedAt time.Time
}

type Session struct {
	ID                SessionID
	TenantID          TenantID
	UserID            UserID
	State             FSMState
	Version           int
	OrchestrationTier OrchestrationTier
	Channel           string
	ExpiresAt         time.Time
	ContextPolicy     json.RawMessage
	Metadata          json.RawMessage
	CreatedAt         time.Time
	UpdatedAt         time.Time
}

type Message struct {
	ID           MessageID
	TenantID     TenantID
	SessionID    SessionID
	Role         string
	Content      string
	CitationRefs []string
	TokenCount   int
	JobID        *JobID
	Metadata     json.RawMessage
	CreatedAt    time.Time
}

type Document struct {
	ID              DocumentID
	TenantID        TenantID
	S3Key           string
	FileName        string
	ContentType     string
	Version         int
	IngestionStatus IngestionStatus
	Metadata        json.RawMessage
	CreatedAt       time.Time
	UpdatedAt       time.Time
}

type Chunk struct {
	ID         ChunkID
	TenantID   TenantID
	DocumentID DocumentID
	Ordinal    int
	Content    string
	SourceLoc  string
	Vector     []float32
	TsVector   string
	Metadata   json.RawMessage
	CreatedAt  time.Time
}

type Artifact struct {
	ID        ArtifactID
	TenantID  TenantID
	SessionID SessionID
	S3Key     string
	Type      string
	Metadata  json.RawMessage
	CreatedAt time.Time
}

type PromptVersionRecord struct {
	ID        int64
	Name      PromptName
	Version   PromptVersion
	Content   string
	Metadata  json.RawMessage
	CreatedAt time.Time
}

type AuditLog struct {
	ID              int64
	TenantID        TenantID
	UserID          UserID
	SessionID       SessionID
	ToolName        ToolName
	IdempotencyKey  IdempotencyKey
	RequestHash     string
	ResponseHash    string
	ApprovalRecord  json.RawMessage
	Metadata        json.RawMessage
	CreatedAt       time.Time
}

type RiverJob struct {
	ID             JobID
	TenantID       TenantID
	SessionID      SessionID
	IdempotencyKey IdempotencyKey
	Kind           string
	Payload        json.RawMessage
	Attempt        int
	MaxAttempts    int
	State          string
	Error          *string
	ScheduledAt    time.Time
	CreatedAt      time.Time
	UpdatedAt      time.Time
}

type LLMRequest struct {
	Provider     LLMProvider
	Model        string
	Messages     []LLMMessage
	Temperature  float32
	MaxTokens    int
	JSONSchema   json.RawMessage
	Fallbacks    []LLMProvider
	CostPolicy   CostPolicy
}

type LLMMessage struct {
	Role    string
	Content string
}

type LLMResponse struct {
	Provider       LLMProvider
	Model          string
	Content        string
	StructuredJSON json.RawMessage
	Usage          TokenUsage
	CostUSD        float64
}

type TokenUsage struct {
	PromptTokens     int
	CompletionTokens int
	TotalTokens      int
}

type CostPolicy struct {
	MaxCostUSD   float64
	PreferredModel string
}

type RetrievalQuery struct {
	TenantID      TenantID
	Query         string
	TopK          int
	HybridWeight  float32
	Filter        map[string]any
}

type RetrievalResult struct {
	ChunkID    ChunkID
	DocumentID DocumentID
	DocumentName string
	Ordinal    int
	Content    string
	SourceLoc  string
	Score      float64
}

type RAGContext struct {
	Results   []RetrievalResult
	Citations []Citation
}

type Citation struct {
	ChunkID      ChunkID
	DocumentName string
	Ordinal      int
	SourceLoc    string
}

type ToolSchema struct {
	Name        ToolName
	Description string
	Permission  ToolPermission
	InputSchema json.RawMessage
	OutputSchema json.RawMessage
	CacheTTL    *time.Duration
	RequiresApproval bool
}

type ToolCall struct {
	Name      ToolName
	Arguments json.RawMessage
}

type ToolResult struct {
	Result  json.RawMessage
	Error   error
	Cached  bool
}

type GraphNode struct {
	ID          string
	Name        string
	Kind        string
	Timeout     time.Duration
	Dependencies []string
	Config      json.RawMessage
}

type Graph struct {
	ID      string
	TenantID TenantID
	Nodes   []GraphNode
	Edges   map[string][]string
}

type GraphExecution struct {
	GraphID   string
	SessionID SessionID
	State     map[string]any
	Results   map[string]any
}

type RateLimitConfig struct {
	TenantID   TenantID
	Window     time.Duration
	MaxRequests int
}

type IdempotentOperation interface {
	IdempotencyKey() IdempotencyKey
}

type ModelRouter interface {
	Route(ctx context.Context, req LLMRequest) (LLMResponse, error)
}

type PromptRegistry interface {
	Load(ctx context.Context, tenantID TenantID, name PromptName, version PromptVersion) (string, error)
	Register(ctx context.Context, tenantID TenantID, name PromptName, version PromptVersion, content string) error
}

type StructuredOutputParser interface {
	Parse(ctx context.Context, schema json.RawMessage, output string) (json.RawMessage, error)
	SchemaFor(v any) (json.RawMessage, error)
}

type ToolRegistry interface {
	Register(schema ToolSchema, handler ToolHandler) error
	Call(ctx context.Context, call ToolCall, tenantID TenantID) (ToolResult, error)
	List(permission ToolPermission) []ToolSchema
}

type ToolHandler func(ctx context.Context, tenantID TenantID, args json.RawMessage) (ToolResult, error)

type Retriever interface {
	Retrieve(ctx context.Context, query RetrievalQuery) ([]RetrievalResult, error)
}

type ReRanker interface {
	Rank(ctx context.Context, query string, results []RetrievalResult) ([]RetrievalResult, error)
}

type IngestionPipeline interface {
	Ingest(ctx context.Context, tenantID TenantID, reader io.Reader, filename string, contentType string) (DocumentID, error)
	Reingest(ctx context.Context, docID DocumentID) error
}

type SessionManager interface {
	Create(ctx context.Context, tenantID TenantID, userID UserID, tier OrchestrationTier, channel string) (Session, error)
	Load(ctx context.Context, tenantID TenantID, sessionID SessionID) (Session, error)
	Update(ctx context.Context, session Session) (Session, error)
	Expire(ctx context.Context, before time.Time) error
}

type FSM interface {
	Transition(session Session, event FSMBEvent) (FSMState, error)
}

type FSMEvent struct {
	Kind    string
	Payload any
}

type HITLGate interface {
	RequestApproval(ctx context.Context, sessionID SessionID, action WriteAction) error
	Approve(ctx context.Context, sessionID SessionID, approvedBy UserID) error
	Reject(ctx context.Context, sessionID SessionID, rejectedBy UserID) error
}

type WriteAction struct {
	ToolName       ToolName
	IdempotencyKey IdempotencyKey
	Description    string
}

type ChannelAdapter interface {
	Receive(ctx context.Context, payload []byte) (ChannelMessage, error)
	Send(ctx context.Context, tenantID TenantID, channelUserID string, response ChannelResponse) error
}

type ChannelMessage struct {
	TenantID    TenantID
	UserID      UserID
	SessionID   *SessionID
	Text        string
	Metadata    map[string]any
	ReplyMarkup json.RawMessage
}

type ChannelResponse struct {
	Text        string
	Citations   []Citation
	ReplyMarkup json.RawMessage
}

type RateLimiter interface {
	Allow(ctx context.Context, tenantID TenantID) (bool, error)
}

type Guardrails interface {
	ValidateInput(ctx context.Context, input string) error
	EvaluateCritic(ctx context.Context, answer string, sources []RetrievalResult) (CriticResult, error)
}

type CriticResult struct {
	Score           float64
	RecallAtK       float64
	CitationCoverage float64
	Grounded        bool
}

type CostTracker interface {
	Record(ctx context.Context, tenantID TenantID, usage TokenUsage, costUSD float64) error
}

type ObjectStorage interface {
	Put(ctx context.Context, key string, reader io.Reader, size int64, contentType string) error
	Get(ctx context.Context, key string) (io.ReadCloser, error)
	Delete(ctx context.Context, key string) error
}

type Authenticator interface {
	Authenticate(ctx context.Context, token string) (TenantID, UserID, error)
	AuthenticateAPIKey(ctx context.Context, key APIKey) (TenantID, error)
}

type Repository interface {
	WithTx(ctx context.Context, fn func(pgx.Tx) error) error
	Pool() *pgxpool.Pool
}

type Runtime struct{}

func (r *Runtime) ExecuteL0(ctx context.Context, sessionID SessionID, input string) (Message, error) { return Message{}, nil }
func (r *Runtime) ExecuteL1(ctx context.Context, sessionID SessionID, input string) (Message, error) { return Message{}, nil }
func (r *Runtime) ExecuteL2(ctx context.Context, sessionID SessionID, kind string, payload json.RawMessage) (JobID, error) { return "", nil }
func (r *Runtime) ExecuteL3(ctx context.Context, sessionID SessionID, graph Graph) (JobID, error) { return "", nil }

type APIError struct {
	Code    string
	Message string
	Details map[string]any
}

func (e *APIError) Error() string { return "" }

type Worker struct{}

func (w *Worker) Start(ctx context.Context) error { return nil }
func (w *Worker) Stop(ctx context.Context) error { return nil }

type Server struct{}

func (s *Server) Start(ctx context.Context) error { return nil }
func (s *Server) Stop(ctx context.Context) error { return nil }

type Config struct{}

func LoadConfig(path string) (*Config, error) { return nil, nil }
func (c *Config) PostgresDSN() string { return "" }
func (c *Config) RedisAddr() string { return "" }
func (c *Config) ObjectStorageConfig() map[string]string { return nil }
func (c *Config) ServerMode() string { return "" }

type NullString sql.NullString
```
