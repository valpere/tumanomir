package payments

import (
	"context"
)

type ProviderID string

type TransactionID string

type OperationType string

type OperationResult string

type Transaction struct {
	ID       TransactionID
	Provider ProviderID
	Type     OperationType
	Amount   float64
	Currency string
	Metadata map[string]string
}

type OperationResultRecord struct {
	TransactionID TransactionID
	OperationType OperationType
	Result        OperationResult
	Timestamp     int64
	ErrorMessage  string
}

type Provider interface {
	ProcessTransaction(ctx context.Context, tx Transaction) (OperationResultRecord, error)
	ID() ProviderID
	SupportedOperations() []OperationType
}

type Logger interface {
	LogResult(record OperationResultRecord)
	LogError(tx Transaction, err error)
}

type PaymentProcessor struct {
	providers map[ProviderID]Provider
	logger    Logger
}

func NewPaymentProcessor(logger Logger) *PaymentProcessor {
	return nil
}

func (p *PaymentProcessor) RegisterProvider(provider Provider) {}

func (p *PaymentProcessor) Process(ctx context.Context, tx Transaction) (OperationResultRecord, error) {
	return OperationResultRecord{}, nil
