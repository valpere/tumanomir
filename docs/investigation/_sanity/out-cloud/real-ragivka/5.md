```go
package ragivka

import (
	"context"
	"database/sql"
	"time"

	"github.com/google/uuid"
	"github.com/jackc/pgx/v5/pgxpool"
)

type TenantID uuid.UUID
type SessionID uuid.UUID
type DocumentID uuid.UUID
type ArtifactID uuid.UUID
type JobID uuid.UUID
type ToolName string
type PromptName string
type PromptVersion string
type ModelProvider string
type IdempotencyKey string
type OperationHash string

type Tier int

const (
	L0 Tier = iota
	L1
	L2
	L3
)

type FSMState string

const (
	StateActive          FSMState = "Active"
	StateWaitingForHuman FSMState = "WaitingForHuman"
	StateCompleted       FSMState = "Completed"
	StateExpired         FSMState = "Expired"
)

type Permission string

const (
	PermissionRead  Permission = "Read"
	PermissionDraft Permission = "Draft"
	PermissionWrite Permission = "Write"
)

type MessageRole string

const (
	RoleSystem    MessageRole = "system"
	RoleUser      MessageRole = "user"
	RoleAssistant MessageRole = "assistant"
	RoleTool      MessageRole = "tool"
)

type AuditEventType string

const (
	AuditToolExecution AuditEventType = "tool_execution"
	AuditFSMTransition AuditEventType = "fsm_transition"
)

type Session struct {
	ID            SessionID
	TenantID      TenantID
	State         FSMState
	Version       int64
	ContextWindow []Message
	LastActivity  time.Time
	ExpiresAt     *time.Time
	CreatedAt     time.Time
	UpdatedAt     time.Time
}

type Message struct {
	ID        uuid.UUID
	SessionID SessionID
	Role      MessageRole
	Content   string
	Metadata  map[string]any
	CreatedAt time.Time
}

type Document struct {
	ID         DocumentID
	TenantID   TenantID
	Name       string
	SourceURI  string
	Version    int
	RawStorage string
	Status     string
	CreatedAt  time.Time
	UpdatedAt  time.Time
}

type Chunk struct {
	ID         uuid.UUID
	DocumentID DocumentID
	TenantID   TenantID
	Ordinal    int
	Content    string
	SourceLoc  string
	Embedding  []float32
	Tokens     int
	CreatedAt  time.Time
}

type Artifact struct {
	ID           ArtifactID
	TenantID     TenantID
	SessionID    SessionID
	DocumentID   *DocumentID
	StorageURI   string
	ContentType  string
	GeneratedBy  ToolName
	Idempotency  IdempotencyKey
	RequestHash  OperationHash
	ResponseHash OperationHash
	CreatedAt    time.Time
}

type Prompt struct {
	Name       PromptName
	Version    PromptVersion
	Template   string
	Variables  []string
	Registered bool
	CreatedAt  time.Time
}

type Tool struct {
	Name        ToolName
	Description string
	Permission  Permission
	Schema      JSONSchema
	Handler     ToolHandler
	CacheConfig *CacheConfig
}

type CacheConfig struct {
	Enabled bool
	TTL     time.Duration
	KeyFn   func(context.Context, ToolCall) string
}

type ToolCall struct {
	ID        uuid.UUID
	SessionID SessionID
	ToolName  ToolName
	Arguments map[string]any
}

type ToolResult struct {
	CallID  uuid.UUID
	Outcome map[string]any
	Error   *ToolError
}

type ToolError struct {
	Code    string
	Message string
}

type ToolHandler func(context.Context, ToolCall) (ToolResult, error)

type JSONSchema struct {
	Schema map[string]any
}

type LLMRequest struct {
	SessionID   SessionID
	TenantID    TenantID
	Model       ModelProvider
	Messages    []Message
	Tools       []Tool
	Temperature float64
	MaxTokens   int
}

type LLMResponse struct {
	Content      string
	ToolCalls    []ToolCall
	FinishReason string
	Usage        TokenUsage
}

type TokenUsage struct {
	PromptTokens     int
	CompletionTokens int
	TotalTokens      int
	EstimatedCost    float64
}

type ModelRouter interface {
	Route(ctx context.Context, req LLMRequest) (LLMResponse, error)
}

type Retriever interface {
	HybridSearch(ctx context.Context, tenantID TenantID, query string, topK int) ([]SearchResult, error)
}

type SearchResult struct {
	ChunkID      uuid.UUID
	DocumentID   DocumentID
	DocumentName string
	Ordinal      int
	Score        float64
	SourceLoc    string
	Content      string
}

type ReRanker interface {
	Rank(ctx context.Context, query string, results []SearchResult) ([]SearchResult, error)
}

type Citation struct {
	DocumentName string
	ChunkOrdinal int
	SourceLoc    string
	Excerpt      string
}

type RAGResponse struct {
	Answer    string
	Citations []Citation
	Usage     TokenUsage
}

type IngestionPipeline interface {
	Ingest(ctx context.Context, tenantID TenantID, sourceURI string, docType string) (DocumentID, error)
	Reingest(ctx context.Context, docID DocumentID) error
}

type ArtifactGenerator interface {
	GeneratePDF(ctx context.Context, data any, template string) (*Artifact, error)
	GenerateExcel(ctx context.Context, data any) (*Artifact, error)
}

type SessionManager interface {
	Create(ctx context.Context, tenantID TenantID) (SessionID, error)
	Get(ctx context.Context, id SessionID) (*Session, error)
	AppendMessage(ctx context.Context, id SessionID, msg Message) (*Session, error)
	Transition(ctx context.Context, id SessionID, to FSMState) (*Session, error)
	ExpireStale(ctx context.Context, before time.Time) ([]SessionID, error)
	ApplyRetentionPolicy(ctx context.Context, id SessionID, maxTurns int) error
}

type JobSpec struct {
	ID        JobID
	TenantID  TenantID
	SessionID SessionID
	Tier      Tier
	Payload   map[string]any
}

type Worker interface {
	Enqueue(ctx context.Context, spec JobSpec) (JobID, error)
	Claim(ctx context.Context, queue string) (*JobSpec, error)
	Complete(ctx context.Context, id JobID) error
	Fail(ctx context.Context, id JobID, err error) error
}

type RateLimiter interface {
	Allow(ctx context.Context, tenantID TenantID, key string, limit int, window time.Duration) (bool, error)
}

type ToolRegistry interface {
	Register(tool Tool) error
	Get(name ToolName) (Tool, bool)
	List() []Tool
}

type PromptRegistry interface {
	Store(ctx context.Context, prompt Prompt) error
	Load(ctx context.Context, name PromptName, version PromptVersion) (*Prompt, error)
}

type ChannelAdapter interface {
	HandleWebhook(ctx context.Context, payload []byte) error
	SendMessage(ctx context.Context, tenantID TenantID, sessionID SessionID, content string) error
}

type TelegramAdapter struct{}
type WebWidgetAdapter struct{}

type AuthVerifier interface {
	Verify(ctx context.Context, token string) (*TenantID, error)
}

type PIIStripper interface {
	Strip(ctx context.Context, text string) (string, error)
}

type EvalHook interface {
	RecordRetrievalRecall(ctx context.Context, query string, results []SearchResult, expected []uuid.UUID)
	RecordCitationCoverage(ctx context.Context, answer string, citations []Citation)
	RecordGroundedness(ctx context.Context, answer string, context []string) (float64, error)
}

type AuditLogger interface {
	Log(ctx context.Context, eventType AuditEventType, tenantID TenantID, idempotency IdempotencyKey, tool ToolName, reqHash, respHash OperationHash) error
}

type MetricsCollector interface {
	RecordLLMTokenUsage(provider ModelProvider, prompt, completion int)
	RecordRetrievalLatency(d time.Duration)
	RecordQueueDepth(queue string, depth int)
	RecordErrorRate(service string, success bool)
}

type Tracer interface {
	Start(ctx context.Context, name string) (context.Context, Span)
}

type Span interface {
	End()
}

type Config struct {
	APIPort             int
	WorkerCount         int
	RedisAddr           string
	PostgresURL         string
	ObjectStorageURL    string
	OfflineMode         bool
	DefaultEmbeddingDim int
}

type App struct {
	DB             *pgxpool.Pool
	Pool           *pgxpool.Pool
	Retriever      Retriever
	Router         ModelRouter
	Sessions       SessionManager
	Tools          ToolRegistry
	Prompts        PromptRegistry
	Workers        Worker
	RateLimiter    RateLimiter
	ArtifactGen    ArtifactGenerator
	Audit          AuditLogger
	Metrics        MetricsCollector
	Tracer         Tracer
	PII            PIIStripper
	Auth           AuthVerifier
	Eval           EvalHook
	ChannelAdapters []ChannelAdapter
}

func NewApp(cfg Config) (*App, error) { return nil, nil }

func (a *App) RunAPI(ctx context.Context) error { return nil }
func (a *App) RunWorker(ctx context.Context) error { return nil }
func (a *App) RunAll(ctx context.Context) error { return nil }

func (a *App) HandleL0(ctx context.Context, tenantID TenantID, req LLMRequest) (*LLMResponse, error) { return nil, nil }
func (a *App) HandleL1(ctx context.Context, tenantID TenantID, sessionID SessionID, query string) (*RAGResponse, error) { return nil, nil }
func (a *App) HandleL2(ctx context.Context, tenantID TenantID, spec JobSpec) (JobID, error) { return JobID{}, nil }
func (a *App) HandleL3(ctx context.Context, tenantID TenantID, graph DAG) (JobID, error) { return JobID{}, nil }

func (s *SessionManager) ExecInTx(ctx context.Context, fn func(*sql.Tx) error) error { return nil }

type DAG struct {
	Nodes []AgentNode
	Edges []Edge
}

type AgentNode struct {
	ID      uuid.UUID
	Name    string
	Timeout time.Duration
	Execute func(context.Context, AgentState) (AgentState, error)
}

type Edge struct {
	From uuid.UUID
	To   uuid.UUID
}

type AgentState struct {
	Data map[string]any
}

type AgentOrchestrator interface {
	Execute(ctx context.Context, dag DAG, timeout time.Duration) (AgentState, error)
	DetectDeadlock(dag DAG) error
}

type ErrorResponse struct {
	Code    string  `json:"code"`
	Message string  `json:"message"`
	Details []ErrorDetail `json:"details,omitempty"`
}

type ErrorDetail struct {
	Field   string `json:"field"`
	Message string `json:"message"`
}

func NewErrorResponse(code, message string, details []ErrorDetail) ErrorResponse { return ErrorResponse{} }
```
