package ragivka

import (
	"context"
	"database/sql"
	"io"
	"time"
)

type TenantID string
type SessionID string
type MessageID string
type DocumentID string
type ArtifactID string
type JobID string
type IdempotencyKey string
type ToolName string
type PromptName string
type PromptVersion string
type ModelProvider string
type OperationKey string
type TraceID string

const (
	ModelProviderOpenAI     ModelProvider = "openai"
	ModelProviderAnthropic  ModelProvider = "anthropic"
	ModelProviderOpenRouter ModelProvider = "openrouter"
	ModelProviderGemini     ModelProvider = "gemini"
	ModelProviderOllama     ModelProvider = "ollama"
)

const (
	FSMStateActive         string = "active"
	FSMStateWaitingForHuman string = "waiting_for_human"
	FSMStateCompleted      string = "completed"
	FSMStateExpired        string = "expired"
)

const (
	PermissionRead  string = "read"
	PermissionDraft string = "draft"
	PermissionWrite string = "write"
)

const (
	OrchestrationL0 string = "L0"
	OrchestrationL1 string = "L1"
	OrchestrationL2 string = "L2"
	OrchestrationL3 string = "L3"
)

type Tenant struct {
	ID        TenantID
	Name      string
	CreatedAt time.Time
}

type Session struct {
	ID          SessionID
	TenantID    TenantID
	State       string
	Version     int
	LastMessage time.Time
	CreatedAt   time.Time
	ExpiresAt   *time.Time
	Metadata    map[string]string
}

type Message struct {
	ID        MessageID
	SessionID SessionID
	TenantID  TenantID
	Role      string
	Content   string
	Turn      int
	CreatedAt time.Time
}

type Citation struct {
	DocumentID   DocumentID
	DocumentName string
	ChunkOrdinal int
	SourceLocation string
	ChunkID      string
}

type RAGResult struct {
	ChunkID      string
	DocumentID   DocumentID
	Content      string
	Score        float64
	SourceLocation string
	Ordinal      int
}

type LLMRequest struct {
	TenantID     TenantID
	SessionID    SessionID
	TraceID      TraceID
	PromptName   PromptName
	PromptVersion PromptVersion
	Messages     []Message
	Temperature  float32
	MaxTokens    int
	JSONSchema   map[string]interface{}
}

type LLMResponse struct {
	Content      string
	Structured   map[string]interface{}
	Provider     ModelProvider
	Model        string
	PromptTokens int
	CompletionTokens int
	TotalTokens  int
	EstimatedCost float64
}

type ToolArgument struct {
	Name  string
	Value interface{}
}

type ToolCall struct {
	ID        string
	ToolName  ToolName
	Arguments []ToolArgument
}

type ToolResult struct {
	CallID   string
	ToolName ToolName
	Output   interface{}
	Error    error
}

type Tool struct {
	Name        ToolName
	Permission  string
	Schema      map[string]interface{}
	Handler     ToolHandler
	CacheTTL    *time.Duration
}

type ToolHandler func(ctx context.Context, args []ToolArgument) (interface{}, error)

type ToolRegistry interface {
	Register(tool Tool) error
	Get(name ToolName) (Tool, error)
	List() []Tool
}

type ModelRouter interface {
	Route(ctx context.Context, req LLMRequest) (LLMResponse, error)
	RegisterProvider(provider ModelProvider, client LLMClient) error
}

type LLMClient interface {
	Complete(ctx context.Context, req LLMRequest) (LLMResponse, error)
}

type PromptRegistry interface {
	Get(name PromptName, version PromptVersion) (Prompt, error)
	Latest(name PromptName) (Prompt, error)
	Store(prompt Prompt) error
}

type Prompt struct {
	Name        PromptName
	Version     PromptVersion
	Template    string
	Variables   []string
	CreatedAt   time.Time
}

type Retriever interface {
	Retrieve(ctx context.Context, tenantID TenantID, query string, topK int) ([]RAGResult, error)
}

type ReRanker interface {
	ReRank(ctx context.Context, query string, results []RAGResult, topK int) ([]RAGResult, error)
}

type Chunk struct {
	ID           string
	DocumentID   DocumentID
	Content      string
	Embedding    []float32
	Ordinal      int
	SourceLocation string
	Metadata     map[string]string
}

type Document struct {
	ID          DocumentID
	TenantID    TenantID
	Name        string
	ContentType string
	Size        int64
	ObjectKey   string
	Version     int
	CreatedAt   time.Time
	UpdatedAt   time.Time
}

type Artifact struct {
	ID        ArtifactID
	TenantID  TenantID
	SessionID SessionID
	Kind      string
	ObjectKey string
	CreatedAt time.Time
}

type IngestionJob struct {
	JobID      JobID
	TenantID   TenantID
	SourceURI  string
	DocumentID DocumentID
	Status     string
}

type Connector interface {
	Ingest(ctx context.Context, tenantID TenantID, sourceURI string) (Document, error)
}

type Parser interface {
	Parse(ctx context.Context, doc Document) ([]Chunk, error)
}

type ObjectStorage interface {
	Put(ctx context.Context, key string, reader io.Reader, size int64, contentType string) error
	Get(ctx context.Context, key string) (io.ReadCloser, error)
	Delete(ctx context.Context, key string) error
}

type RateLimiter interface {
	Allow(ctx context.Context, tenantID TenantID, key string, limit int, window time.Duration) (bool, error)
}

type Evaluator interface {
	RecallAtK(ctx context.Context, results []RAGResult, relevant []string, k int) (float64, error)
	CitationCoverage(ctx context.Context, answer string, citations []Citation) (float64, error)
	GroundednessHook(ctx context.Context, claim string, context []string) (float64, error)
}

type AuditLogger interface {
	Log(ctx context.Context, entry AuditLogEntry) error
}

type AuditLogEntry struct {
	ID             string
	TenantID       TenantID
	IdempotencyKey IdempotencyKey
	ToolName       ToolName
	SessionID      SessionID
	Action         string
	RequestHash    string
	ResponseHash   string
	CreatedAt      time.Time
}

type SessionManager interface {
	Create(ctx context.Context, tenantID TenantID) (Session, error)
	Get(ctx context.Context, sessionID SessionID) (Session, error)
	AppendMessage(ctx context.Context, sessionID SessionID, role, content string) error
	Transition(ctx context.Context, sessionID SessionID, fromState, toState string) error
	ApplyRetentionPolicy(ctx context.Context, sessionID SessionID, maxTurns int) error
	ExpireInactive(ctx context.Context, before time.Time) error
}

type Orchestrator interface {
	ExecuteL0(ctx context.Context, req LLMRequest) (LLMResponse, error)
	ExecuteL1(ctx context.Context, sessionID SessionID, userMessage string) (AssistantResponse, error)
	ExecuteL2(ctx context.Context, sessionID SessionID, workflow WorkflowDefinition) (JobID, error)
	ExecuteL3(ctx context.Context, sessionID SessionID, graph GraphDefinition) (JobID, error)
}

type AssistantResponse struct {
	SessionID  SessionID
	State       string
	Content     string
	Citations   []Citation
	ToolCalls   []ToolCall
	Artifacts   []Artifact
}

type WorkflowDefinition struct {
	Name  string
	Steps []WorkflowStep
}

type WorkflowStep struct {
	Name      string
	Action    string
	DependsOn []string
	Input     map[string]interface{}
}

type GraphDefinition struct {
	Name  string
	Nodes []GraphNode
	Edges []GraphEdge
}

type GraphNode struct {
	ID      string
	Kind    string
	Timeout time.Duration
}

type GraphEdge struct {
	From string
	To   string
}

type Worker interface {
	Start(ctx context.Context) error
	Stop(ctx context.Context) error
	Enqueue(ctx context.Context, job Job) (JobID, error)
}

type Job struct {
	ID       JobID
	TenantID TenantID
	Queue    string
	Payload  []byte
}

type ChannelAdapter interface {
	HandleWebhook(ctx context.Context, payload []byte) error
	SendMessage(ctx context.Context, tenantID TenantID, recipient, content string) error
}

type TelegramAdapter struct{}

type WebWidgetAdapter struct{}

type APIError struct {
	Code    string
	Message string
	Details map[string]interface{}
}

type Auth interface {
	Authenticate(ctx context.Context, token string) (TenantID, error)
	Authorize(ctx context.Context, tenantID TenantID, permission string) error
}

type PIIStripper interface {
	Strip(ctx context.Context, text string) (string, error)
}

type CostTracker interface {
	Record(ctx context.Context, tenantID TenantID, req LLMRequest, resp LLMResponse) error
}

type MetricsCollector interface {
	RecordLLMTokens(ctx context.Context, tenantID TenantID, prompt, completion int)
	RecordRetrievalLatency(ctx context.Context, tenantID TenantID, duration time.Duration)
	RecordQueueDepth(ctx context.Context, queue string, depth int)
	RecordError(ctx context.Context, tenantID TenantID, code string)
}

type Tracer interface {
	StartSpan(ctx context.Context, name string) (context.Context, Span)
}

type Span interface {
	End()
	SetError(err error)
	SetAttribute(key string, value interface{})
}

type DB interface {
	Pool() interface{}
	Begin(ctx context.Context) (Tx, error)
}

type Tx interface {
	Commit(ctx context.Context) error
	Rollback(ctx context.Context) error
	Exec(ctx context.Context, sql string, args ...interface{}) (sql.Result, error)
	Query(ctx context.Context, sql string, args ...interface{}) (Rows, error)
}

type Rows interface {
	Next() bool
	Scan(dest ...interface{}) error
	Close() error
}
