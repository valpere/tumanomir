package ragivka

import (
	"context"
	"database/sql"
	"io"
	"time"

	"github.com/gin-gonic/gin"
	"github.com/google/uuid"
	"github.com/jackc/pgx/v5/pgxpool"
	"github.com/riverqueue/river"
)

// ─── Named Types / Constants ─────────────────────────────────────────────────

type TenantID uuid.UUID
type SessionID uuid.UUID
type MessageID uuid.UUID
type DocumentID uuid.UUID
type ChunkID uuid.UUID
type ArtifactID uuid.UUID
type ToolName string
type PromptName string
type ProviderName string
type OperationKey string

type Tier string

const (
	TierL0 Tier = "L0"
	TierL1 Tier = "L1"
	TierL2 Tier = "L2"
	TierL3 Tier = "L3"
)

type FSMState string

const (
	FSMStateActive          FSMState = "Active"
	FSMStateWaitingForHuman FSMState = "WaitingForHuman"
	FSMStateCompleted       FSMState = "Completed"
	FSMStateExpired         FSMState = "Expired"
)

type ToolPermission string

const (
	ToolPermissionRead  ToolPermission = "Read"
	ToolPermissionDraft ToolPermission = "Draft"
	ToolPermissionWrite ToolPermission = "Write"
)

type DeploymentMode string

const (
	DeploymentSingleBinary DeploymentMode = "single-binary"
	DeploymentCompose      DeploymentMode = "docker-compose"
	DeploymentSplit        DeploymentMode = "split-api-worker"
	DeploymentOffline      DeploymentMode = "offline"
)

// ─── Configuration ───────────────────────────────────────────────────────────

type Config struct {
	DeploymentMode   DeploymentMode
	PostgreSQLURL    string
	RedisURL         string
	ObjectStorageURL string
	APIListenAddr    string
	WorkerCount      int
	OfflineMode      bool
	DefaultTenantRateLimit RateLimitConfig
}

type RateLimitConfig struct {
	RequestsPerMinute int
	Burst             int
}

// ─── Domain Models ───────────────────────────────────────────────────────────

type Tenant struct {
	ID        TenantID
	Name      string
	CreatedAt time.Time
}

type Session struct {
	ID           SessionID
	TenantID     TenantID
	State        FSMState
	Version      int
	Context      []Message
	LastActivity time.Time
	ExpiresAt    time.Time
	CreatedAt    time.Time
	UpdatedAt    time.Time
}

type Message struct {
	ID        MessageID
	SessionID SessionID
	Role      string
	Content   string
	TurnIndex int
	CreatedAt time.Time
}

type Document struct {
	ID          DocumentID
	TenantID    TenantID
	Name        string
	SourceURI   string
	Version     int
	RawStoredAt string
	IngestedAt  time.Time
}

type Chunk struct {
	ID          ChunkID
	TenantID    TenantID
	DocumentID  DocumentID
	Ordinal     int
	TokenStart  int
	TokenEnd    int
	Text        string
	Vector      []float32
	Tsvector    string
	SourcePage  int
	CreatedAt   time.Time
}

type Citation struct {
	DocumentName string
	ChunkOrdinal int
	SourcePage   int
}

type Artifact struct {
	ID        ArtifactID
	TenantID  TenantID
	SessionID SessionID
	Kind      string
	ObjectKey string
	Hash      string
	CreatedAt time.Time
}

type AuditLog struct {
	ID             uuid.UUID
	TenantID       TenantID
	IdempotencyKey OperationKey
	ToolName       ToolName
	Action         string
	RequestHash    string
	ResponseHash   string
	FSMFrom        FSMState
	FSMTo          FSMState
	Actor          string
	CreatedAt      time.Time
}

// ─── LLM / AI Layer ───────────────────────────────────────────────────────────

type ModelRequest struct {
	Provider    ProviderName
	Model       string
	Messages    []Message
	Temperature float64
	MaxTokens   int
	JSONSchema  any
}

type ModelResponse struct {
	Content      string
	FinishReason string
	Usage        TokenUsage
	CostUSD      float64
}

type TokenUsage struct {
	PromptTokens     int
	CompletionTokens int
	TotalTokens      int
}

type ModelRouter interface {
	Route(ctx context.Context, task TaskComplexity, req ModelRequest) (ModelResponse, error)
	Fallback(ctx context.Context, req ModelRequest, err error) (ModelResponse, error)
}

type PromptRegistry interface {
	Load(ctx context.Context, name PromptName, version string) (Prompt, error)
}

type Prompt struct {
	Name    PromptName
	Version string
	Content string
}

type TaskComplexity string

const (
	TaskClassification TaskComplexity = "classification"
	TaskExtraction     TaskComplexity = "extraction"
	TaskReasoning      TaskComplexity = "reasoning"
)

// ─── RAG ─────────────────────────────────────────────────────────────────────

type IngestRequest struct {
	TenantID   TenantID
	SourceURI  string
	DocumentID DocumentID
	Format     string
}

type SearchRequest struct {
	TenantID   TenantID
	Query      string
	TopK       int
	UseHybrid  bool
	Rerank     bool
}

type SearchResult struct {
	Chunk    Chunk
	Score    float64
	Rank     int
	Citation Citation
}

type IngestionPipeline interface {
	Ingest(ctx context.Context, req IngestRequest) (DocumentID, error)
	Parse(ctx context.Context, r io.Reader, format string) (string, error)
	Chunk(ctx context.Context, docID DocumentID, text string) ([]Chunk, error)
	Index(ctx context.Context, chunks []Chunk) error
}

type HybridRetriever interface {
	Search(ctx context.Context, req SearchRequest) ([]SearchResult, error)
}

type Reranker interface {
	Rerank(ctx context.Context, query string, results []SearchResult) ([]SearchResult, error)
}

// ─── Tools ───────────────────────────────────────────────────────────────────

type Tool interface {
	Name() ToolName
	Permission() ToolPermission
	Schema() any
	Execute(ctx context.Context, args json.RawMessage) (any, error)
}

type ToolRegistry interface {
	Register(t Tool) error
	Resolve(name ToolName) (Tool, bool)
}

type FunctionCall struct {
	Name      ToolName
	Arguments json.RawMessage
}

type HITLGate struct {
	SessionID SessionID
	Reason    string
	ToolName  ToolName
	Proposed  any
}

// ─── Orchestration ───────────────────────────────────────────────────────────

type Orchestrator interface {
	RunL0(ctx context.Context, tenantID TenantID, prompt PromptName, input string) (string, error)
	RunL1(ctx context.Context, tenantID TenantID, sessionID SessionID, userMessage string) (*Session, []FunctionCall, error)
	RunL2(ctx context.Context, tenantID TenantID, workflow string, payload any) (*river.Job, error)
	RunL3(ctx context.Context, tenantID TenantID, dag DAG) error
}

type DAG struct {
	TenantID TenantID
	SessionID SessionID
	Nodes     []Node
	Edges     []Edge
	Timeout   time.Duration
}

type Node struct {
	ID       string
	Role     string
	Prompt   PromptName
	Children []string
	Timeout  time.Duration
}

type Edge struct {
	From string
	To   string
}

// ─── State Machine / Session Manager ───────────────────────────────────────────

type SessionManager interface {
	Create(ctx context.Context, tenantID TenantID) (SessionID, error)
	Append(ctx context.Context, sessionID SessionID, msg Message) (*Session, error)
	Transition(ctx context.Context, sessionID SessionID, from, to FSMState, version int) (*Session, error)
	Expire(ctx context.Context, before time.Time) ([]SessionID, error)
	PruneHistory(ctx context.Context, sessionID SessionID, maxTurns int) error
}

// ─── Channel Adapters ────────────────────────────────────────────────────────

type TelegramAdapter interface {

}

// ─── Worker / Jobs ───────────────────────────────────────────────────────────
